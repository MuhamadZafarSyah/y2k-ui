# Separator

Flat divider lines for grouping content in Y2K panels and menus.

<ComponentPreview
  name="separator-demo"
  preview={<SeparatorDemo />}
  source={`import { Separator } from "@/components/ui/separator"

export function SeparatorDemo() {
  return (
    <div>
      <span>Top</span>
      <Separator className="my-3" variant="pink" />
      <span>Bottom</span>
    </div>
  )
}`}
/>

## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui add separator" />

## Usage

```tsx
import { Separator } from "@/components/ui/separator"

export function Example() {
  return <Separator variant="pink" />
}
```
