# Card

A Y2K flat panel — thick navy border, pastel title bar, white content area.
Use it as the canvas for grouped information.

<ComponentPreview
  name="card-default-demo"
  preview={<CardDefaultDemo />}
  source={`import {
  Card,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from "@/components/ui/card"

export function CardDefaultDemo() {
  return (
    <Card>
      <CardHeader>
        <CardTitle>Welcome back</CardTitle>
        <CardDescription>Pick up where you left off</CardDescription>
      </CardHeader>
      <CardContent>You have 3 unread messages.</CardContent>
      <CardFooter>
        <span>Last seen 2h ago</span>
        <button>Open</button>
      </CardFooter>
    </Card>
  )
}`}
/>

import {

  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from "@/components/ui/card";
