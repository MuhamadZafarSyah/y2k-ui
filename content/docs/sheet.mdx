# Sheet

A slide-in panel with a Y2K window title bar. Animates from any edge of the screen.

## Installation

<CodeBlock
  language="bash"
  filename="terminal.sh"
  code="npx y2kui add sheet"
/>

## Usage

```tsx
import {
  Sheet,
  SheetContent,
  SheetDescription,
  SheetHeader,
  SheetTitle,
  SheetTrigger,
} from "@/components/ui/sheet";

<Sheet>
  <SheetTrigger asChild>
    <Button>Open</Button>
  </SheetTrigger>
  <SheetContent side="right" title="panel.app">
    <SheetHeader>
      <SheetTitle>Panel Title</SheetTitle>
      <SheetDescription>Panel description here.</SheetDescription>
    </SheetHeader>
    {/* Content */}
  </SheetContent>
</Sheet>;
```
