## Installation

<CodeBlock
  language="bash"
  filename="terminal.sh"
  code="npx y2kui@latest add button"
/>

## Usage

```tsx
import { Button } from "@/components/ui/button";

export function Example() {
  return <Button variant="pink">Click me</Button>;
}
```
