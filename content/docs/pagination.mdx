# Pagination

A Y2K-styled pagination component featuring thick navy outlines, pastel highlight on the active page, monospace numbers, and Previous/Next buttons with arrow icons.

<Pagination>
  <PaginationContent>
    <PaginationItem>
      <PaginationPrevious href="#" />
    </PaginationItem>
    <PaginationItem>
      <PaginationLink href="#">1</PaginationLink>

    <PaginationItem>
      <PaginationLink href="#" isActive>

    </PaginationItem>
    <PaginationItem>
      <PaginationLink href="#">4</PaginationLink>

    <PaginationItem>

    </PaginationItem>
    <PaginationItem>
      <PaginationNext href="#" />

  </PaginationContent>
</Pagination>`}
/>

## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add pagination" />

## Usage

```tsx
import {
  Pagination,
  PaginationContent,
  PaginationEllipsis,
  PaginationItem,
  PaginationLink,
  PaginationNext,
  PaginationPrevious,
} from "@/components/ui/pagination"
```
