## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add item" />

## Usage

```tsx
import { Item, ItemLabel, ItemIndicator } from "@/components/ui/item"

<Item selected>
  <ItemIndicator>✓</ItemIndicator>
  <ItemLabel>Selected option</ItemLabel>
</Item>
```
