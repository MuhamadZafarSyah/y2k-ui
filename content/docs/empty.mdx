# Empty

An empty state placeholder for when no data is available.

  name="empty-demo"
  preview={<EmptyDemo />}

<Empty
  icon={<SearchIcon />}
  title="No results found"
  description="Try adjusting your search terms or filters."
  action={<Button size="sm">Clear filters</Button>}
/>`}
/>

## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add empty" />

## Usage

```tsx
import { Empty } from "@/components/ui/empty"

<Empty
  icon={<InboxIcon />}
  title="No messages"
  description="You have no unread messages."
  action={<Button>New message</Button>}
/>
```
