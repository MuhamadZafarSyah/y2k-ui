# Direction Provider

Wraps your app to set text direction for RTL or LTR layouts.

<ComponentPreview
  name="direction-demo"
  preview={<DirectionDemo />}
  source={`import { DirectionProvider } from "@/components/ui/direction"

<DirectionProvider dir="rtl">
  <YourApp />
</DirectionProvider>`}
/>

## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add direction" />

## Usage

```tsx
import { DirectionProvider } from "@/components/ui/direction"

<DirectionProvider dir="rtl">
  <YourApp />
</DirectionProvider>
```
