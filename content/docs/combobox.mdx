# Combobox

A searchable select built on Radix UI Popover and cmdk. The popup opens below the trigger and includes a live search input.

<ComponentPreview
  name="combobox-demo"
  preview={<ComboboxDemo />}
  source={`import {
  Combobox,
  ComboboxTrigger,
  ComboboxValue,
  ComboboxPopup,
  ComboboxItem,
} from "@/components/ui/combobox"

<Combobox defaultValue="apple">
  <ComboboxTrigger className="w-48">
    <ComboboxValue />
  </ComboboxTrigger>
  <ComboboxPopup>
    <ComboboxItem value="apple">Apple</ComboboxItem>
    <ComboboxItem value="banana">Banana</ComboboxItem>
    <ComboboxItem value="cherry">Cherry</ComboboxItem>
  </ComboboxPopup>
</Combobox>`}
/>

## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add combobox" />

## Usage

```tsx
import {
  Combobox,
  ComboboxTrigger,
  ComboboxValue,
  ComboboxPopup,
  ComboboxItem,
} from "@/components/ui/combobox"

<Combobox defaultValue="apple">
  <ComboboxTrigger className="w-48">
    <ComboboxValue />
  </ComboboxTrigger>
  <ComboboxPopup>
    <ComboboxItem value="apple">Apple</ComboboxItem>
    <ComboboxItem value="banana">Banana</ComboboxItem>
    <ComboboxItem value="cherry">Cherry</ComboboxItem>
  </ComboboxPopup>
</Combobox>
```
