# Calendar

A date picker calendar built on react-day-picker.

<ComponentPreview
  name="calendar-demo"
  preview={<CalendarDemo />}
  source={`"use client"
import { Calendar } from "@/components/ui/calendar"
import { useState } from "react"

export function CalendarDemo() {
  const [date, setDate] = useState<Date | undefined>(new Date())
  return <Calendar selected={date} onSelect={setDate} mode="single" />
}`}
/>

## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add calendar" />

## Usage

```tsx
"use client"
import { Calendar } from "@/components/ui/calendar"
import { useState } from "react"

export function Example() {
  const [date, setDate] = useState<Date | undefined>(new Date())
  return <Calendar selected={date} onSelect={setDate} mode="single" />
}
```
