## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add native-select" />

## Usage

```tsx
import { NativeSelect } from "@/components/ui/native-select"

<NativeSelect>
  <option>Option 1</option>
  <option>Option 2</option>
</NativeSelect>
```
