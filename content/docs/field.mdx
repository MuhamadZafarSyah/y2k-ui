# Field

A layout wrapper for form controls with label, description, and error message slots.

<ComponentPreview
  name="field-demo"
  preview={<FieldDemo />}
  source={`import { Field, FieldLabel, FieldControl, FieldDescription } from "@/components/ui/field"
import { Input } from "@/components/ui/input"

<Field>
  <FieldLabel htmlFor="email">Email</FieldLabel>
  <FieldControl>
    <Input id="email" type="email" placeholder="you@example.com" />
  </FieldControl>
  <FieldDescription>We'll never share your email.</FieldDescription>
</Field>`}
/>

## Installation

<CodeBlock language="bash" filename="terminal.sh" code="npx y2kui@latest add field" />

## Usage

```tsx
import { Field, FieldLabel, FieldControl, FieldDescription, FieldError } from "@/components/ui/field"

<Field>
  <FieldLabel htmlFor="email">Email</FieldLabel>
  <FieldControl>
    <Input id="email" />
  </FieldControl>
  <FieldDescription>We'll never share your email.</FieldDescription>
  <FieldError>Email is required.</FieldError>
</Field>
```
